Olá!!! bom vou postar aki um código d como eu faço::


Código:

#include <allegro.h>



int main()
{
   allegro_init();
   install_keyboard();
   set_color_depth(16);
   set_gfx_mode(GFX_AUTODETECT_WINDOWED, 640, 480, 0, 0);

   BITMAP *imgSprites = NULL;
   imgSprites = load_bitmap("robocop.bmp",NULL);

   BITMAP *Buffer = NULL;
   Buffer = create_bitmap(SCREEN_W,SCREEN_H);

   const int iSprite_Largura = 69;   // Largura de um sprite
   const int iSprite_Altura = 80;     // Altura de um sprite
   const int Qtde_Spritesx = 6;      // Qtde de Desenhos em x[0] - Horizontal
   const int Qtde_Spritesy = 0;      // Qtde de Desenhos em y[0] - Vertical

   const int iCiclo = 30; // Ciclo de troca de sprite

   int iPosx = 0; // Posição x do corte.
   int iPosy = 0; // Posição y do corte. Como é apenas uma fileira então y sempre vai ser 0

   int buffer_animacao = 0; // Buffer para controle do ciclo

   while (!key[KEY_ESC])
   {
      clear_bitmap( Buffer );
      textprintf_ex(Buffer, font, 0, 0, makecol(0, 255, 255),-1,"ANIMACAO SPRITES");

      // Controle para mudar o sprite
      if(buffer_animacao == 0)
      {
            if (iPosx > 0 && iPosx == Qtde_Spritesx-1)
            {
               iPosx = 0;
            }

            if (Qtde_Spritesx > 0)
            {
               iPosx++;
            }

            buffer_animacao = iCiclo;

      }
      else
      {
         buffer_animacao--;
      }

      // Corta e desenha o sprite
      masked_blit(imgSprites,Buffer,iPosx*iSprite_Largura,iPosy,SCREEN_W/2, SCREEN_H/2,iSprite_Largura,iSprite_Altura);

      blit(Buffer,screen,0,0,0,0,SCREEN_W,SCREEN_H);
      vsync();

   }

   destroy_bitmap(imgSprites);
   destroy_bitmap(Buffer);   

   allegro_exit();

   return 0;
}
END_OF_MAIN();

Você carrega um BITMAP com os sprites, vamos supor que os sprites tem 20x20, e o bitmap tem 4 sprites, todos na mesma linha, então o bitmap principal tem 80x20, certo?
Código:
BITMAP *main = load_bitmap("sprites.bmp", NULL);

Agora você cria o seu array e cada elemento vai conter um sprite, você faz assim:
Código:
BITMAP *sprites[4];

int i;
for (i = 0; i < 4; i++)
{
   sprites[i] = create_sub_bitmap(main, i*20, 0, 20, 20);
}

A função create_sub_bitmap é definida assim:
Código:
BITMAP *create_sub_bitmap(BITMAP *parent, int x, y, width, height);

E a vantagem dela é que não está Copiando uma parte do bitmap principal para os outros, e sim criando uma referência a eles! Então você não tem que se preocoupar com a memória que eles vão utilizar, basta destruir o bitmap Principal no fim! =D
