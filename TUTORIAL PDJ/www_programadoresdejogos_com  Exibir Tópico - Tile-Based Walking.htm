Bom nao manjo de c++ e nem de alegro.
Mais pra isso se pode faze a movimentacao dele se do tamanho dos tiles.
Tipo seu tile eh 32x32 entao pra movimenta o player se pode faze assim:

boneco_x += velocidade_boneco vezes 32 que seria o tamanho do tile.

Ow voce podi cria duas variaveis dx e dy do personagem.
E ai voce inicia elas sendo igual ao valor x e y do personagem.
Vo tenta da um exemplo usando isso que falei
Código:

Se aperto pra direita
dx := dx + 32;
Se aperto pra cima
dy := dy + 32;

Ai no timer se poe assim:

Se boneco_x < dx entao
boneco_x += velocidade_boneco

Se boneco_x > dx entao
boneco_x -= velocidade_boneco

E mema coisa pro dy

void movimentar()
{
/*
Função que dita a direção do boneco e faz o boneco andar para o próximo tile.
1 - cima; 2 - baixo; 3 - direita; 4 - esquerda; 5 - cima-direita;
6 - cima-esquerda; 7 - baixo-direita; 8 - baixo-esquerda;
*/
  if (direcao == 1)
  {
    if (boneco_y > prox_tile_y)
    {
        boneco_y -= velocidade_boneco;
    }
    else
    {
    andando = 0;
    }
  }
  if (direcao == 2)
  {
    if (boneco_y < prox_tile_y)
    {
        boneco_y += velocidade_boneco;
    }
    else
    {
    andando = 0;
    }
  }
  if (direcao == 3)
  {
    if (boneco_x < prox_tile_x)
    {
        boneco_x += velocidade_boneco;
    }
    else
    {
    andando = 0;
    }
  }
  if (direcao == 4)
  {
    if (boneco_x > prox_tile_x)
    {
        boneco_x -= velocidade_boneco;
    }
    else
    {
    andando = 0;
    }
  }
  if (direcao == 5)
  {
    if ((boneco_x < prox_tile_x) && (boneco_y > prox_tile_y))
    {
        boneco_x += velocidade_boneco;
        boneco_y -= velocidade_boneco;
    }
    else
    {
    andando = 0;
    }
  }
  if (direcao == 6)
  {
    if ((boneco_x > prox_tile_x) && (boneco_y > prox_tile_y))
    {
        boneco_x -= velocidade_boneco;
        boneco_y -= velocidade_boneco;
    }
    else
    {
    andando = 0;
    }
  }
  if (direcao == 7)
  {
    if ((boneco_x < prox_tile_x) && (boneco_y < prox_tile_y))
    {
        boneco_x += velocidade_boneco;
        boneco_y += velocidade_boneco;

        boneco_x -= velocidade_boneco;
        boneco_y += velocidade_boneco;
    }
    else
    {
    andando = 0;
    }
  }
}

void movimento_player () // FUNÇÃO PARA MOVIMENTAR O BONECO. (TILE-BASED)
{
        if (andando == 0) {
              if(key[KEY_RIGHT])// DIREITA (AUMENTA X)
              {

                     //boneco_y -= velocidade_boneco;
                     prox_tile_x = boneco_x + 32;
                     prox_tile_y = boneco_y - 32;

                     movimentar();

                 {
                     prox_tile_x = boneco_x - 32;

                     //boneco_x -= velocidade_boneco;
                     //boneco_y += velocidade_boneco;
                     prox_tile_x = boneco_x - 32;
                     prox_tile_y = boneco_y + 32;
                     andando = 1;
                     direcao = 8;
                     movimentar();
                     }
                     else
                     {
                     //boneco_x -= velocidade_boneco;
                     prox_tile_x = boneco_x - 32;
                     prox_tile_y = boneco_y;

                     movimentar();
                     }
                 }
              }

                 if ((boneco_y - 32) >= 0)
                 {
                    prox_tile_x = boneco_x;
                    prox_tile_y = boneco_y - 32;
                    andando = 1;
                    direcao = 1;
                    movimentar();
                 }
              }

              {
                 //boneco_y += velocidade_boneco;

        movimentar();
        }
